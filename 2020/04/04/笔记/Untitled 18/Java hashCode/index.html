Java中的hashCode

如果调用系统的hashCode 的时候，需要覆盖hashCode()方法和equals() 方法，否则会因两个对象的地址不同而产生不同的hashCode

package HashCode;

/**
 * @author quicysver
 * @version 1.0.0
 * @ClassName Student.java
 * @Description TODO
 * @createTime 2020年03月09日 16:23:00
 */
public class Student {

    int grade;
    int cls;
    String firstName;
    String lastName;

    public Student(int grade, int cls, String firstName, String lastName) {
        this.grade = grade;
        this.cls = cls;
        this.firstName = firstName;
        this.lastName = lastName;
    }

    @Override
    public int hashCode() {
        int B = 31;

        int hash = 0;
        hash = hash * B + grade;
        hash = hash * B + cls;
        //整型溢出时仍然得到一个整型，不做处理
        hash = hash * B + firstName.toLowerCase().hashCode();
        hash = hash * B + lastName.toLowerCase().hashCode();
        //调用java.util的hashCode()方法

        return hash;
    }

    public boolean equals(Object o) {
        if (this == o)
            return true;

        if (o == null)
            return false;

        if (getClass() != o.getClass())
            return false;

        Student another = (Student) o;

        //开始比较
        return this.grade == another.grade && this.cls == another.cls && this.firstName.toLowerCase()
                .equals(((Student) o).firstName.toLowerCase()) && this.lastName.toLowerCase()
                .equals(((Student) o).lastName.toLowerCase());
    }
}
